#学员系统
‘’’初始化界面
=============================学员管理系统=============================
1 查看学员信息 2 添加学员信息
3 删除学员信息 4 退出系统
=====================================================================
请输入对应选择

需求分析：六个功能
1 存储学员信息源
2 查看学院信息
3 添加学员信息
4 删除学员信息
5 退出
6 交互界面

‘’’
#建立数据源

stu_list=[{‘name’:’罗云熙’,’age’:16,’classid’:’china’},
{‘name’:’ivy’,’age’:12,’classid’:’japan’},
{‘name’:’magit’,’age’:12,’classid’:’japan’}]

def show_stu_info():
if len(stu_list)==0:
print(‘+’*20,’没有学员信息’,’+’*20)
return

print('|{0:<5}|{1:<20}|{2:<5}|{3:<20}|'.format('sid','name','age','classid'))
print('-'*50)
for i,stu_dic in enumerate(stu_list):
    print('|{0:<5}|{1:<20}|{2:<5}|{3:<20}|'.format(i+1,stu_dic['name'],stu_dic['age'],stu_dic['classid']))

#3添加学员
def add_stu(name,age,classid):
stu_dict={}
stu_dict[‘name’]=name
stu_dict[‘age’]=age
stu_dict[‘classid’]=classid
stu_list.append(stu_dict)
#add_stu(‘朱一龙’,18,’qing’)

#show_stu_info()

#4删除学员
def del_stu(sid):
sid_int=int(sid)#强制转换int
stu_list.pop(sid_int-1)

#del_stu(3)
#show_stu_info()

#5退出系统
def log_out():
pass

#界面交互
while True:#死循环，等待用户输入
#初始化界面
print(‘+’*20,’学员管理系统’,’+’*20)
print(‘{:1} {:15} {:15}’.format(‘ ‘,’1. 查看学员信息’,’2. 添加学员信息’))
print(‘{:1} {:15} {:15}’.format(‘ ‘,’2. 删除学员信息’,’2. 退出’))
print(‘+’*50)
key=input(‘请输入对应的选择’)

if key=='1':
    print('+' * 20, '学员信息浏览', '+' * 20)
    show_stu_info()
    input('按回车返回：')
elif key=='2':
    print('+' * 20, '添加学员信息', '+' * 20)
    name=input('请输入姓名')
    age=input('请输入年龄')
    classid=input('请输入学号')
    add_stu(name,age,classid)
    show_stu_info()
    input('按回车返回：')
elif key=='3':
    print('+' * 20, '删除学员信息', '+' * 20)
    show_stu_info()
    sid=input("请输入要删除的学员的sid")
    del_stu(sid)
    show_stu_info()
    input('按回车返回：')
elif key=='4':
    log_out()
    print('+' * 20, '再见', '+' * 20)
    break
else:
    print('+' * 20, '无效输入', '+' * 20)
