class Student(object):
def init(self,name,score):

self.score=score
def print_student(self):

#加下划线表示私有不能修改
class Student1(object):
def init(self,name,score):
self.__name=name
self.__score=score
def print_student(self):
print(‘%s:%s’%(self.__name,self.__score))
def get_grade(self):
if self.__score>=90:
return ‘A’
elif self.__score>=60:
return ‘B’
else:
return ‘C’
def set_score(self,score):
if score in range(0,101):
self.__score=score
else:
raise ValueError(“请输入0-100之间的分数”)
def get_score(self):
return self.__score

xiaolv=Student1(‘xiaolv’,2)
xiaolv.set_score(109)
