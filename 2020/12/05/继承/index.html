‘’’子类中方法与父类一样的话，会覆盖父类方法，并且可以添加别的方法，可用于更新操作，而不需要动父类’’’
class Dog(Animal):
def run(self):

super().run()
print(“Dog is running”)
def eat(self):

dog.run()
Dog.run
dog.eat()’’’

#gettattr(),setattr(),hasattr()
class MyOb(object):
def init(self):
self.x=9
def power(self):
return self.x*self.x
obj=MyOb()

#判断obj是否有属性’x’
print(hasattr(obj,’x’))#True
print(hasattr(obj,’y’))#False

#设置一个属性y
setattr(obj,’y’,91)
print(hasattr(obj,’y’))#False
print(getattr(obj,’y’))
#dir的应用
print(obj.dir())
print(dir(obj))
