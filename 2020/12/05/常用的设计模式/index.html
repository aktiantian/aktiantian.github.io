#yield的用法，生成斐波那契数列 后一个数永远是前两个数字的和 1 1 2 3 5 8
def fib(max):
n,a,b =0,0,1
while n<max:
print(b,end=’ ‘)
a,b=b,a+b
n=n+1
return ‘done’
fib(6)

#输出关键字换成yield
def fib1(max):
n,a,b =0,0,1
while n<max:
yield b
a,b=b,a+b
n=n+1
return ‘done’

g=fib1(9)
for n in g:
print(n,end=’| ‘)
#用next方法
while True:
try:
x=next(g)
print(x)
except StopIteration as e:
print(‘没有数据了’,e.value)
break
