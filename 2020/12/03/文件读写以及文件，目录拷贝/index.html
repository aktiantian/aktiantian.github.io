import os
def copy_file(file1,file2):
‘’’
文件复制函数
:param file1: 源文件
:param file2: 目标文件
:return:
‘’’’’
#打开源文件和目标文件
f1=open(file1,’rb’)
f2=open(file2,’wb’)
#循环读取写入
content=f1.readline()
while len(content)>0:
f2.write(content)
content=f1.readline()
f1.close()
f2.close()
#copy_file(‘./photo.jpg’,’./judy.jpg’)#./为当前目录

#通过导入os模块复制目录
def copy_path(dir1,dir2):
‘’’
:param dir1: 源目录
:param dir2: 复制目录
:return:
‘’’
# 获取被复制目录中的所有文件，包含了文件和目录
dlist1=os.listdir(dir1)

#创建新目录
os.mkdir(dir2)

#遍历dlist1判断是文件还是目录，并执行文件复制

for f in dlist1:
    #为遍历的文件添加目录路径
    file1=os.path.join(dir1,f)#源文件
    file2=os.path.join(dir2,f)#目标文件
    if os.path.isfile(file1):#判断是不是文件
        copy_file(file1,file2)#调用文件复制函数
    if os.path.isdir(file1):#判判断如果是个目录
        copy_path(file1,file2)#递归调用自己来实现子目录的复制

copy_path(‘./test1’,’./test2’)
